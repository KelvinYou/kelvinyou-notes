# Flatten JSON

One of the coding test during interview

Company: Grab Malaysia


Input

```js
const obj = {
   "one": 1,
   "two": {
      "three": 3
   },
   "four": {
      "five": 5,
      "six": {
         "seven": 7
      },
      "eight": 8
   },
   "nine": 9
};
```
Output

```js
const output = {
   'one': 1,
   'two.three': 3,
   'four.five': 5,
   'four.six.seven': 7,
   'four.eight': 8,
   'nine': 9
};
```

<summary>Answer</summary>

```js
const obj = {
   "one": 1,
   "two": {
      "three": 3
   },
   "four": {
      "five": 5,
      "six": {
         "seven": 7
      },
      "eight": 8
   },
   "nine": 9
};

const flattenJSON = (obj = {}, res = {}, extraKey = '') => {
  for(key in obj){
    if(typeof obj[key] !== 'object'){
      res[extraKey + key] = obj[key];
    }else{
      flattenJSON(obj[key], res, `${extraKey}${key}.`);
    };
  };
  return res;
};

console.log(flattenJSON(obj));
```
